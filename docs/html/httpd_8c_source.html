138 PT_THREAD(enqueue_chunk(httpd_state *s, uint8_t immediate,
139  const char *format, ...))
140 {
141  va_list ap;
142 
143  PSOCK_BEGIN(&s->sout);
144 
145  va_start(ap, format);
146 
147  s->tmp_buf_len = vsnprintf(s->tmp_buf, TMP_BUF_SIZE, format, ap);
148 
149  va_end(ap);
150 
151  if(s->blen + s->tmp_buf_len < HTTPD_SIMPLE_MAIN_BUF_SIZE) {
152  /* Enough space for the entire chunk. Copy over */
153  memcpy(&s->buf[s->blen], s->tmp_buf, s->tmp_buf_len);
154  s->blen += s->tmp_buf_len;
155  } else {
156  memcpy(&s->buf[s->blen], s->tmp_buf, HTTPD_SIMPLE_MAIN_BUF_SIZE - s->blen);
157  s->tmp_buf_copied = HTTPD_SIMPLE_MAIN_BUF_SIZE - s->blen;
158  s->blen = HTTPD_SIMPLE_MAIN_BUF_SIZE;
159  PSOCK_SEND(&s->sout, (uint8_t *)s->buf, s->blen);
160  s->blen = 0;
161  if(s->tmp_buf_copied < s->tmp_buf_len) {
162  memcpy(s->buf, &s->tmp_buf[s->tmp_buf_copied],
163  s->tmp_buf_len - s->tmp_buf_copied);
164  s->blen += s->tmp_buf_len - s->tmp_buf_copied;
165  }
166  }
167 
168  if(immediate != 0 && s->blen > 0) {
169  PSOCK_SEND(&s->sout, (uint8_t *)s->buf, s->blen);
170  s->blen = 0;
171  }
172 
173  PSOCK_END(&s->sout);
174 }

208  PT_BEGIN(&s->generate_pt);

374 static
375 PT_THREAD(handle_input(httpd_state *s))
376 {
377  PRINTF("** Processing Input.\n");
378 
379  PSOCK_BEGIN(&s->sin);
380 
381  PSOCK_READTO(&s->sin, ISO_space);
382 
383  /* ---------------------------- Handle GET and PUT ---------------------------- */
384  if( (strncasecmp(s->inputbuf, http_get, 4) == 0) ||
385  (strncasecmp(s->inputbuf, http_put, 4) == 0)) {
386 
387  if(strncasecmp(s->inputbuf, http_get, 4) == 0) {
388  PRINTF("***** handle_input: GET *******\n");
389  s->request_type = REQUEST_TYPE_GET;
390  } else {
391  PRINTF("***** handle_input: PUT *******\n");
392  s->request_type = REQUEST_TYPE_PUT;
393  }
394 
395  PSOCK_READTO(&s->sin, ISO_space);
396  // copy uri
397  s->complete_uri_len = PSOCK_DATALEN(&s->sin) - 1; // we remove the space at the end
398  memcpy(&s->complete_uri, &s->inputbuf, s->complete_uri_len);
399  s->complete_uri[s->complete_uri_len] = '\0';
400 
401  s->uri = s->complete_uri;
402  s->uri_query = strstr(s->complete_uri, "?");
403 
404  if(s->uri_query != NULL){
405  /* move forward one byte uri_query to remove '?' */
406  s->uri_query++;
407  s->uri_len = s->uri_query - s->complete_uri - 1; /* -1 to remove '?' from complete_uri */
408  s->uri_query_len = s->complete_uri_len - s->uri_len;
409  }
410  else{
411  s->uri_query_len = 0;
412  s->uri_len = s->complete_uri_len;
413  }
414 
415  if(s->inputbuf[0] != ISO_slash) {
416  PSOCK_CLOSE_EXIT(&s->sin);
417  }
418 
419  /* ---------------------------- Handle POST ---------------------------- */
420  } else if(strncasecmp(s->inputbuf, http_post, 5) == 0) {
421 
422  PRINTF("***** handle_input: POST *******\n");
423  s->request_type = REQUEST_TYPE_POST;
424  PSOCK_READTO(&s->sin, ISO_space);
425 
426  if(s->inputbuf[0] != ISO_slash) {
427  PSOCK_CLOSE_EXIT(&s->sin);
428  }
429 
430 
431  s->complete_uri_len = PSOCK_DATALEN(&s->sin) - 1; // we remove the space at the end
432  memcpy(&s->complete_uri, &s->inputbuf, s->complete_uri_len);
433  s->complete_uri[s->complete_uri_len] = '\0';
434 
435  s->uri = s->complete_uri;
436  s->uri_query = strstr(s->complete_uri, "?");
437 
438  if(s->uri_query != NULL){
439  /* move forward one byte uri_query to remove '?' */
440  s->uri_query++;
441  s->uri_len = s->uri_query - s->complete_uri - 1; /* -1 to remove '?' from complete_uri */
442  s->uri_query_len = s->complete_uri_len - s->uri_len;
443  }
444  else{
445  s->uri_query_len = 0;
446  s->uri_len = s->complete_uri_len;
447  }
448 
449  //s->inputbuf[PSOCK_DATALEN(&s->sin) - 1] = ISO_null_term;
450 
451 
452  /* POST: Read out the rest of the line and ignore it */
453  PSOCK_READTO(&s->sin, ISO_nl);
454 
455  /*
456  * Start parsing headers. We look for Content-Length and ignore everything
457  * else until we hit the start of the message body.status_code
458  *
459  * We will return 411 if the client doesn't send Content-Length and 413
460  * if Content-Length is too high
461  */
462  s->content_length = 0;
463  s->return_code = RETURN_CODE_LR;
464 
465  do {
466  s->inputbuf[PSOCK_DATALEN(&s->sin)] = 0;
467  /* We anticipate a content length */
468  if((PSOCK_DATALEN(&s->sin) > 14) &&
469  strncasecmp(s->inputbuf, "Content-Length:", 15) == 0) {
470  char *val_start = &s->inputbuf[15];
471  s->content_length = atoi(val_start);
472 
473  /* So far so good */
474  s->return_code = RETURN_CODE_OK;
475  }
476  PSOCK_READTO(&s->sin, ISO_nl);
477  } while(PSOCK_DATALEN(&s->sin) != 2);
478 
479  /*
480  * Done reading headers.
481  * Reject content length greater than CONTENT_LENGTH_MAX bytes
482  */
483  if(s->content_length > CONTENT_LENGTH_MAX) {
484  s->content_length = 0;
485  s->return_code = RETURN_CODE_TL;
486  }
487 
488  /* Parse the message body, unless we have detected an error. */
489 
490  s->response.content_length = s->content_length;
491  ptr_toWrite = s->buffer;
492  PRINTF("content_length = %d || return_code = %d\n",
493  s->content_length,
494  s->return_code);
495 
496  while(s->content_length > 0 /*&& lock == s */&&
497  s->return_code == RETURN_CODE_OK) {
498  PSOCK_READBUF_LEN(&s->sin, s->content_length);
499 
500  s->content_length -= PSOCK_DATALEN(&s->sin);
501 
502  int len = PSOCK_DATALEN(&s->sin);
503  memcpy(ptr_toWrite, &s->inputbuf, len);
504  ptr_toWrite += len;
505 
506  }
507 
508  // TODO: retirar isto e ver se dá barraca...
509  if(s->content_length > 0)
510  *(++ptr_toWrite) = ISO_null_term;
511 
512  //
513  s->content_length = s->response.content_length;
514  s->response.content_length = 0;
515  /* ---------------------------- Handle PUT ---------------------------- */
516  } else {
517  PSOCK_CLOSE_EXIT(&s->sin);
518  }
519 
520  s->state = STATE_OUTPUT;
521 
522  while(1) {
523  PSOCK_READTO(&s->sin, ISO_nl);
524  }
525 
526  PSOCK_END(&s->sin);
527 }

539 handle_connection(httpd_state *s)
540 {
541  int res_found = 0;
542 
543  if(s->state & STATE_IGNORE) {
544  return;
545  }
546 
547  PRINTF("** Handle Connection\n");
548  if(!(s->state & STATE_OUTPUT)) {
549  handle_input(s);
550  }
551 
552  if(s->state & STATE_OUTPUT) {
553  /* By default, all endpoints have an immediate response */
554  s->response.immediate_response = 1;
555 
556  /* have already call service callback for this request? */
557  if(!(s->state & STATE_PROCESSED)) {
558 
559  /* call service callback only if received data is correct */
560  if(s->return_code == RETURN_CODE_OK) {
561  PRINTF("** Calling Service Callback\n");
562  //
563  res_found = service_cbk(s, &s->response, (uint8_t *)&s->response.buf, 0, 0);
564  // Update the request state
565  if(res_found) {
566  s->state |= STATE_RES_FOUND;
567  }
568  } else {
569  PRINTF("** Received data NOK!\n");
570  /* Force Resource Found to make handle_output process the error */
571  s->state |= STATE_RES_FOUND;
572  }
573 
574  s->state |= STATE_PROCESSED;
575 
576  PRINTF("\n** State: %d, res_found = %d, Immediate: %d\n", s->return_code, res_found, s->response.immediate_response);
577  } else {
578  PRINTF("** Request already processed or CoAP response: %d\n", s->state);
579  }
580 
581  //
582  if(s->response.immediate_response){
583  handle_output(s);
584  }
585  }

687 static void
688 appcall(void *state)
689 {
690  httpd_state *s = (httpd_state *)state;
691 
692  if(uip_closed() || uip_aborted() || uip_timedout()) {
693  if(s != NULL) {
694  reset_http_state_obj(s);
695  memb_free(&conns, s);
696  }
697  } else if(uip_connected()) {
698  s = (httpd_state *)memb_alloc(&conns);
699  if(s == NULL) {
700  uip_abort();
701  return;
702  }
703 
704  reset_http_state_obj(s);
705  tcp_markconn(uip_conn, s);
706  PSOCK_INIT(&s->sin, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
707  PSOCK_INIT(&s->sout, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
708 
709  PT_INIT(&s->outputpt);
710 
711  s->state = STATE_WAITING;
712 
713  timer_set(&s->timer, CLOCK_SECOND * ABORT_CONN_TIMEOUT);
714  handle_connection(s);
715 
716  } else if(s != NULL) {
717  if(uip_poll()) {
718  if(timer_expired(&s->timer)) {
719  uip_abort();
