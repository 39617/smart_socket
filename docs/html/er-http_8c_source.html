460 static int
461 http_get_variable(const char *buffer, size_t length, const char *name,
462  const char **output)
463 {
464  const char *start = NULL;
465  const char *end = NULL;
466  const char *value_end = NULL;
467  size_t name_len = 0;
468 
469  /*initialize the output buffer first */
470  *output = 0;
471 
472  name_len = strlen(name);
473  end = buffer + length;
474 
475  for(start = buffer; start + name_len < end; ++start) {
476  if((start == buffer || start[-1] == '&') && start[name_len] == '='
477  && strncmp(name, start, name_len) == 0) {
478 
479  /* Point start to variable value */
480  start += name_len + 1;
481 
482  /* Point end to the end of the value */
483  value_end = (const char *)memchr(start, '&', end - start);
484  if(value_end == NULL) {
485  value_end = end - 1;
486  }
487  *output = start;
488 
489  return value_end - start;
490  }
491  }
492  return 0;
493 }
